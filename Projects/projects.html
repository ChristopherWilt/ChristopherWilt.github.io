public float AlphaBeta(Node n, int depth, float a, float b, bool isMax) {
    if (depth == 0 || n.IsTerminal()) {
        return CalcAdaptiveHeuristic(n);
    }

    if (isMax) {
        float maxEval = float.NegativeInfinity;
        foreach (Node child in n.GetValidMoves()) {
            float eval = AlphaBeta(child, depth - 1, a, b, false);
            maxEval = Mathf.Max(maxEval, eval);
            a = Mathf.Max(a, eval);
            
            // Mathematical Branch Cutoff
            if (b <= a) break; // Prune
        }
        return maxEval;
    }
}
